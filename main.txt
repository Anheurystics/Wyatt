vert basic `
#version 130

in vec3 pos;

void main() {
	gl_Position = vec4(pos, 1.0);
}
`;

frag basic `#version 130

out vec4 FragColor;

void main() {
	gl_FragColor = vec4(1.0, 1.0, 1.0, 1.0);
}
`;

func init() {
	allocate a;
	a.pos <- [-1,-1,0], [1,-1,0], [0,1,0];
}

func loop() {
	use basic;
	draw a;
}
